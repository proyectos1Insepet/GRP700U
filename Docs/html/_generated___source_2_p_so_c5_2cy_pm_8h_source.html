173 /*******************************************************************************
174 * This marco is used to obtain the CPU frequency in MHz. It should be only used
175 * when the clock distribution system is prepared for the low power mode entry.
176 * This macro is silicon dependent as PSoC 5 devices have no CPU clock divider
177 * and PSoC 3 devices have different placement of the CPU clock divider register
178 * bitfield.
179 *******************************************************************************/
180 #if(CY_PSOC3)
181  #define CY_PM_GET_CPU_FREQ_MHZ \
182  ((uint32)(cyPmImoFreqReg2Mhz[CY_PM_FASTCLK_IMO_CR_REG & CY_PM_FASTCLK_IMO_CR_FREQ_MASK]) / \
183  ((uint8)(((CY_PM_CLKDIST_MSTR1_REG & CY_PM_CLKDIST_CPU_DIV_MASK) >> 4u) + 1u)))
184 #endif /* (CY_PSOC3) */
185 
186 #if(CY_PSOC5)

196 * power modes in PSoC 5 devices are invoked by Wait-For-Interrupt (WFI)
197 * instruction. The ARM compilers has __wfi() intrinsic that inserts a WFI
198 * instruction into the instruction stream generated by the compiler. The GCC
199 * compiler has to execute assembly language instruction.
200 *******************************************************************************/

202

204  #define CY_PM_WFI __wfi()

206  #define CY_PM_WFI __asm volatile ("WFI \n")
207  #endif /* (__ARMCC_VERSION) */
208 
209 #else
